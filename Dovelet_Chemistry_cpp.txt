#include <iostream>
#include <stdio.h>
#include <cmath>
using namespace std;
int main() {
       float pret, aftt;
       cout << fixed;
       cout.precision(2);
       for (int i = 0; i != -1; i++) {
              if (i == 0) {
                     cin >> pret;
              }
              cin >> aftt;
              if (aftt == 999) {
                     cout << "End of Output";
                     return 0;
              }
              cout << aftt - pret << endl;
              pret = aftt;
       }
       return 0;
}
